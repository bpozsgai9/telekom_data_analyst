SELECT NET_SERVICE_NAME, (COUNT(*) / (SELECT COUNT(*) FROM contract)) * 100 AS '%'
FROM contract
WHERE SEGMENT_GROUP = 'LAK'
GROUP BY NET_SERVICE_NAME
ORDER BY SEGMENT_GROUP DESC;

SELECT NET_SERVICE_NAME, (COUNT(*) / (SELECT COUNT(*) FROM contract)) * 100 AS '%'
FROM contract
WHERE SEGMENT_GROUP = 'ENT'
GROUP BY NET_SERVICE_NAME
ORDER BY SEGMENT_GROUP DESC;
